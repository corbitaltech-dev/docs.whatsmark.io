<Info>
  **Scope required:** `contacts:write`
</Info>

Update a contact. All fields optional; `phone` uniqueness still enforced.

<ParamField path="contact_id" type="string" required>
  Identifier of the contact.
</ParamField>

<ParamField body="first_name" type="string">
  The contact's first name. Max length 100.
</ParamField>

<ParamField body="last_name" type="string">
  The contact's last name. Max length 100.
</ParamField>

<ParamField body="phone" type="string">
  Must be unique within your workspace. Max length 20.
</ParamField>

<ParamField body="email" type="string">
  The contact's email address. Max length 255.
</ParamField>

<ParamField body="display_name" type="string">
  The contact's display name. Max length 255.
</ParamField>

<ParamField body="type" type="string">
  One of `guest`, `lead`, or `customer`.
</ParamField>

<ParamField body="contact_status_id" type="integer">
  Must reference an existing `contact_statuses` record.
</ParamField>

<ParamField body="contact_source_id" type="integer">
  Must reference an existing `contact_sources` record.
</ParamField>

<ParamField body="is_active" type="boolean">
  Whether the contact is active.
</ParamField>

<ParamField body="group_ids" type="integer[]">
  Each id must reference an existing `contact_groups` record.
</ParamField>

<ParamField body="label_ids" type="integer[]">
  Each id must reference an existing `contact_labels` record.
</ParamField>

<ParamField body="custom_fields" type="array">
  Custom field values.
</ParamField>

<ResponseField name="data" type="object">
  The updated contact.

  <Expandable title="properties">
                  
  </Expandable>
</ResponseField>

<RequestExample>

```bash Request
curl --request PUT \
  --url https://api.whatsmark.io/api/v1/contacts/1 \
  --header "Authorization: Bearer YOUR_API_TOKEN" \
  --header "Content-Type: application/json" \
  --data '{
    "first_name": "Jane",
    "email": "jane.doe@example.com",
    "contact_status_id": 2,
    "label_ids": [1, 2]
  }'
```

</RequestExample>
