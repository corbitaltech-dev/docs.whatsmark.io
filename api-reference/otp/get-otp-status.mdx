---
api: "GET /api/v1/otp/{reference}"
playground: "simple"
title: "Get OTP Status"
description: "Returns the OTP's current status and remaining verification attempts."
---

<Info>
  **Scope required:** `otp:verify`
</Info>

Returns the OTP's current `status` (`pending` / `verified` / `expired` / `cancelled`) and `remaining_attempts`.

<ParamField path="reference" type="string" required>
  Identifier of the reference.
</ParamField>

<ResponseField name="data" type="object">
  <Expandable title="properties">
    <ResponseField name="reference" type="string">
      Opaque identifier for this OTP.
    </ResponseField>

    <ResponseField name="phone" type="string">
      The recipient's phone number.
    </ResponseField>

    <ResponseField name="channel_id" type="integer">
      The channel the OTP was sent from.
    </ResponseField>

    <ResponseField name="purpose" type="string">
      The purpose label for this OTP.
    </ResponseField>

    <ResponseField name="status" type="string">
      The OTP's current status.
    </ResponseField>

    <ResponseField name="expires_at" type="string">
      Timestamp when the OTP expires.
    </ResponseField>

    <ResponseField name="verified_at" type="string">
      Timestamp when the OTP was verified, or `null` if not yet verified.
    </ResponseField>

    <ResponseField name="verification_attempts" type="integer">
      Number of verification attempts made so far.
    </ResponseField>

    <ResponseField name="remaining_attempts" type="integer">
      Number of verification attempts remaining.
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Timestamp when the OTP was created.
    </ResponseField>
  </Expandable>
</ResponseField>
